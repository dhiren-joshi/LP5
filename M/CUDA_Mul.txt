%%writefile matrix_mul.cu
#include <iostream>
#include <cuda_runtime.h>

using namespace std;

// CUDA Kernel
__global__ void matrixMul(int *A, int *B, int *C, int n) {

    int row = threadIdx.y;
    int col = threadIdx.x;

    int sum = 0;

    for (int k = 0; k < n; k++) {
        sum += A[row * n + k] * B[k * n + col];
    }

    C[row * n + col] = sum;
}

int main() {

    int n;

    cout << "Enter size of matrix: ";
    cin >> n;

    int size = n * n * sizeof(int);

    int *A = new int[n * n];
    int *B = new int[n * n];
    int *C = new int[n * n];

    cout << "Enter elements of Matrix A:\n";

    for (int i = 0; i < n * n; i++)
        cin >> A[i];

    cout << "Enter elements of Matrix B:\n";

    for (int i = 0; i < n * n; i++)
        cin >> B[i];

    int *d_A, *d_B, *d_C;

    // Allocate GPU memory
    cudaMalloc(&d_A, size);
    cudaMalloc(&d_B, size);
    cudaMalloc(&d_C, size);

    // Copy data to GPU
    cudaMemcpy(d_A, A, size, cudaMemcpyHostToDevice);
    cudaMemcpy(d_B, B, size, cudaMemcpyHostToDevice);

    // Define threads
    dim3 threads(n, n);

    // Launch Kernel
    matrixMul<<<1, threads>>>(d_A, d_B, d_C, n);

    // Copy result back
    cudaMemcpy(C, d_C, size, cudaMemcpyDeviceToHost);

    cout << "\nResult Matrix:\n";

    for (int i = 0; i < n; i++) {

        for (int j = 0; j < n; j++) {
            cout << C[i * n + j] << " ";
        }

        cout << endl;
    }

    // Free memory
    cudaFree(d_A);
    cudaFree(d_B);
    cudaFree(d_C);

    delete[] A;
    delete[] B;
    delete[] C;

    return 0;
}
