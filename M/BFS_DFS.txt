#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <omp.h>

using namespace std;

vector<vector<int>> graph;
vector<int> visited;

// Parallel BFS Function
void parallelBFS(int start, int n) {

    fill(visited.begin(), visited.end(), 0);

    queue<int> q;
    q.push(start);
    visited[start] = 1;

    cout << "\nParallel BFS: ";

    while (!q.empty()) {

        int node = q.front();
        q.pop();

        cout << node << " ";

        #pragma omp parallel for
        for (int i = 0; i < graph[node].size(); i++) {

            int neigh = graph[node][i];

            if (!visited[neigh]) {

                #pragma omp critical
                {
                    if (!visited[neigh]) {
                        visited[neigh] = 1;
                        q.push(neigh);
                    }
                }
            }
        }
    }

    cout << endl;
}

// Parallel DFS Function
void parallelDFS(int start, int n) {

    fill(visited.begin(), visited.end(), 0);

    stack<int> st;
    st.push(start);

    cout << "Parallel DFS: ";

    while (!st.empty()) {

        int node = st.top();
        st.pop();

        if (!visited[node]) {

            visited[node] = 1;

            cout << node << " ";

            #pragma omp parallel for
            for (int i = 0; i < graph[node].size(); i++) {

                int neigh = graph[node][i];

                if (!visited[neigh]) {

                    #pragma omp critical
                    {
                        st.push(neigh);
                    }
                }
            }
        }
    }

    cout << endl;
}

int main() {

    int n, e;

    cout << "Enter number of vertices: ";
    cin >> n;

    cout << "Enter number of edges: ";
    cin >> e;

    graph.resize(n);
    visited.resize(n, 0);

    cout << "Enter edges (u v):\n";

    for (int i = 0; i < e; i++) {

        int u, v;
        cin >> u >> v;

        // Undirected Graph
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    int start;

    cout << "Enter starting vertex: ";
    cin >> start;

    parallelBFS(start, n);
    parallelDFS(start, n);

    return 0;
}
